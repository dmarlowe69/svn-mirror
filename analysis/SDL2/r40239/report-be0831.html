1371void mmcreplay_io1_store(uint16_t addr, uint8_t value)
1372{
1373 if (rr_active) {
1374 switch (addr & 0xff) {
1375 case 0:
1376 /* $DE00: RR control register write
1377 *
1378 * bit 0: GAME line (W)
1379 * bit 1: EXROM line (W)
1380 * bit 2: 1 = disable RR (W) (*)
1381 * bit 3: bank address 13 (W)
1382 * bit 4: bank address 14 (W)
1383 * bit 5: 0 = rom enable, 1 = ram enable (W)
1384 * bit 6: 1 = exit freeze mode (W)
1385 * bit 7: bank address 15 (W)
1386 * (*) bit can be reset by setting bit 6 of $df12
1387 */
1388
1389 enable_game = value & 1; /* bit 0 */
1390 enable_exrom = ((value >> 1) & 1) ^ 1; /* bit 1 FIXME: is the bit inverted in the register ?! */
1391 bank_address_13_15 = (((value >> 3) & 3) | ((value >> 5) & 4)); /* bit 3,4,7 */
1392 rr_active_bit = value & 4; /* bit 2 */
1393 if (rr_active_bit) {
1394 rr_active = 0;
1395 }
1396 enable_ram_io = (value >> 5) & 1; /* bit 5 */
1397 enable_freeze_exit = (value >> 6) & 1; /* bit 6 */
1398#ifdef LOG_WRITE_DE00
1399 LOG(("MMCREPLAY: IO1 ST %04x %02x enable_game %x enable_exrom %x disable freezer regs %x bank_address_13_15 %x enable_ram_io %x enable_freeze_exit %x", addr, value, enable_game, enable_exrom, value & 4, bank_address_13_15, enable_ram_io, enable_freeze_exit));
1400#endif
1401 mmcreplay_update_mapper(CMODE_WRITE1, enable_freeze_exit);
1402 return;
1403
1404 case 1:
1405 /*
1406 * $DE01: extended RR control register write
1407 * --------------------------------------
1408 * bit 0: 0 = disable clockport, 1 = enable clockport (W)
1409 * (*2) bit 1: 0 = disable io ram banking, 1 = enable io ram banking (W)
1410 * (*2) bit 2: 0 = enable freeze, 1 = disable freeze (W)
1411 * bit 3: bank address 13 (mirror of $DE00) (W)
1412 * bit 4: bank address 14 (mirror of $DE00) (W)
1413 * (*2) bit 5: 0 = enable MMC registers, 1 = disable MMC registers (W)(*1)
1414 * (*2) bit 6: 0 = ram/rom @ DFxx, 1 = ram/rom @ $DExx (W)
1415 * bit 7: bank address 15 (mirror of $DE00) (W)
1416 *
1417 * (*1) Can only be written when bit 6 of $DF12 is 1. Register becomes effective
1418 * when bit 0 of $DF11 is 1.
1419 * (*2) these bits are write-once if bit 0 of $DF11 is 1
1420 */
1421
1422 /* Every bit in $de01 can always be altered in Super Mapper mode+MMC Bios Mode. */
1423 bank_address_13_15 = (((value >> 3) & 3) | ((value >> 5) & 4)); /* bit 3,4,7 */
1424 if (mmcr_clockport_enabled != (value & 1)) {
1425 mmcr_clockport_enabled = value & 1; /* bit 0 */
1426 }

1431 no_freeze = (value >> 2) & 1; /* bit 2 */
1432 enable_ram_io1 = (value >> 6) & 1; /* bit 6 */
1433 if ((enable_extended_mode == 1)
1434 || (enable_16k_mapping == 1)) {
1435 /* FIXME: documentation says that this bit becomes effective when df11 bit0 is 1 */
1436 /*bit 5: 0 = enable MMC registers, 1 = disable MMC registers (W) */
1437 enable_mmc_regs_pending = ((value >> 5) & 1) ^ 1; /* bit 5 */
1438 if (disable_mmc_bios == 1) {
1439 enable_mmc_regs = ((value >> 5) & 1) ^ 1; /* bit 5 */
1440 }
1441 }
1442 if (disable_mmc_bios != 0) {
1443 write_once = 1;
1444 }
1445 }
1446#ifdef LOG_WRITE_DE01
1447 LOG(("MMCREPLAY: IO1 ST %04x %02x mmcr_clockport_enabled %x allow_bank %x no_freeze %x bank_address_13_15 %x enable_mmc_regs_pending %x enable_ram_io1 %x", addr, value, mmcr_clockport_enabled, allow_bank, no_freeze, bank_address_13_15, enable_mmc_regs_pending, enable_ram_io1));
1448#endif
1449 mmcreplay_update_mapper(CMODE_WRITE1, 0);
1450 return;
1451
1452 default:
1453 /*
1454 * $DE02 - $DE0F: Clockport memory area (when enabled)
1455 */
1456 if (mmcr_clockport_enabled) {
1457 if ((addr & 0xff) < 0x10) {
1458#ifdef LOG_CLOCKPORT
1459 LOG(("MMCREPLAY: Clockport ST %04x %02x", addr, value));
1460#endif
1461 return;
1462 }
1463 }
1464 break;
1465 }
1466
1467 if (enable_io1) {
1468 if (enable_ram_io) {
1469#ifdef LOG_WRITE_IO1_RAM
1470 LOG(("store RAM IO1 %04x %02x (%02x:%04x)", addr, value,
1471 io1_ram_bank, (addr & 0x1fff)));
1472#endif
1473 mmcr_ram[(io1_ram_bank << 13) + 0x1e00 + (addr & 0xff)] =
1474 value;
1475 } else {
1476#ifdef LOG_WRITE_IO1_ROM
1477 LOG(("store DISABLED RAM IO1 %04x %02x (%02x:%04x)", addr,
1478 value, io1_ram_bank, (addr & 0x1fff)));
1479#endif
1480 flash040core_store(flashrom_state, (io1_ram_bank << 13) + 0x1e00 + (addr & 0xff), value);
1481 }
1482 } else {
1483#ifdef LOG_WRITE_IO1_DISABLED
1484 LOG(("store DISABLED IO1 %04x %02x (%02x:%04x)", addr, value,
1485 io1_bank, (addr & 0x1fff)));
1486#endif
1487 }
1488 } else { /* rr active */
1489 /* FIXME: when RR regs are disabled, is the mapped RAM still there ? */
1490 if (enable_io1) {
1491 if (enable_ram_io) {
1492#ifdef LOG_WRITE_IO1_RAM
1493 LOG(("store RAM IO1 %04x %02x (%02x:%04x)", addr, value,

1545 * bit 1: 0 = card selected, 1 = card not selected (R/W) (**)
1546 * bit 2: 0 = 250khz transfer, 1 = 8mhz transfer (R/W)
1547 * bit 3: ** ALWAYS 0 **
1548 * bit 4: ** ALWAYS 0 **
1549 * bit 5: 0 = allow RR rom when MMC BIOS disabled , 1 = disable RR ROM (R/W) (***)
1550 * bit 6: 0 = SPI write trigger mode, 1 = SPI read trigger mode (R/W)
1551 * bit 7: ** ALWAYS 0 **
1552 *
1553 * (*) Enabling MMC Bios sets ROM banking to the last 64K bank
1554 * (**) This bit also controls the green activity LED.
1555 * (***) When in mmcreplay bios mode, bit 5 controls RAM banking (0 = $e000 - $ffff, 1 = $8000 - $9fff)
1556 * When in 16K mode, bit 5 enables RAM at $a000 - $bfff
1557 */
1558 if (enable_mmc_regs) {
1559 mmcreplay_io2_device.io_source_valid = 1;
1560 value = disable_mmc_bios; /* bit 0 */
1561 value |= (spi_mmc_card_selected_read() << 1); /* bit 1 */
1562 value |= (spi_mmc_enable_8mhz_read() << 2); /* bit 2 */
1563 /* bit 3,4 always 0 */
1564 value |= (disable_rr_rom << 5); /* bit 5 */
1565 value |= (spi_mmc_trigger_mode_read() << 6); /* bit 6 */
1566 /* bit 7 always 0 */
1567#ifdef LOG_READ_DF11
1568 LOG(("MMCREPLAY: IO2 RD %04x %02x disable_mmc_bios %x disable_rr_rom %x", addr, value, disable_mmc_bios, disable_rr_rom));
1569#endif
1570 return value;
1571 }
1572 break;
1573 case 0x12:
1574 /*
1575 * $DF12: MMC status register
1576 * -----------------------
1577 * bit 0: 0 = SPI ready, 1 = SPI busy (R)
1578 * 1 = forbid ROM write accesses (W) (*)
1579 * bit 1: feedback of $DE00 bit 0 (GAME) (R)
1580 * bit 2: feedback of $DE00 bit 1 (EXROM) (R)
1581 * bit 3: 0 = card inserted, 1 = no card inserted (R)
1582 * bit 4: 0 = card write enabled, 1 = card write disabled (R)
1583 * bit 5: EEPROM DATA line / DDR Register (R/W) (**)
1584 * bit 6: 0 = RR compatibility mode, 1 = Extended mode (W) (***)
1585 * bit 7: EEPROM CLK line (W)
1586 *
1587 * (*) Setting this bit will disable writes to ROM until next reset.
1588 * (**) Setting DATA to "1" enables reading data bit from EEPROM at this position.
1589 * (***) Selecting RR compatibility mode limits RAM to 32K and
1590 * disables writes to extended banking register.
1591 * Selecting Extended mode enables full RAM banking and enables
1592 * Nordic Power mode in RR mode.
1593 */
1594 if (enable_mmc_regs) {
1595 mmcreplay_io2_device.io_source_valid = 1;
1596
1597 value = 0;
1598 /* EEPROM is only accessible in MMC Replay Bios mode */
1599 if (disable_mmc_bios == 0) {
1600 value = eeprom_data_read() << 5; /* bit 5 */
1601 }
1602 value |= (spi_mmc_busy()); /* bit 0 */
1603 value |= (enable_game << 1); /* bit 1 */
1604 value |= (enable_exrom ^ 1) << 2; /* bit 2 FIXME: inverted in reg ? */
1605 value |= (spi_mmc_card_inserted() ^ 1) << 3; /* bit 3 */
1606 value |= (spi_mmc_card_write_enabled() ^ 1) << 4; /* bit 4 */
1607 /* bit 6,7 not readable */
1608#ifdef LOG_READ_DF12
1609 LOG(("MMCREPLAY: IO2 RD %04x %02x enable_game %x enable_exrom %x", addr, value, enable_game, (enable_exrom ^ 1)));
1610#endif
1611 return value;
1612 }
1613 break;
1614 case 0x13:
1615 /*
1616 * $DF13: Extended banking register (*)
1617 * -----------------------------
1618 * bit 0: bank address 16 (R/W)
1619 * bit 1: bank address 17 (R/W)
1620 * bit 2: bank address 18 (R/W)
1621 * bit 3: ** ALWAYS 0 **
1622 * bit 4: ** ALWAYS 0 **
1623 * bit 5: 16K rom mapping (R/W)
1624 * bit 6: 1 = enable RR register (W)
1625 * bit 7: ** ALWAYS 0 **
1626 *
1627 * (*) Can only be read/written to when bit 6 of $DF12 is 1.
1628 */
1629
1630 if (enable_extended_mode) {
1631 mmcreplay_io2_device.io_source_valid = 1;
1632
1633 value = bank_address_16_18; /* bit 0-2 */
1634 /* bit 3,4 always 0 */
1635 value |= enable_16k_mapping << 5; /* bit 5 */
1636 /* bit 7 always 0 */
1637#ifdef LOG_READ_DF13
1638 LOG (("MMCREPLAY: IO2 RD %04x %02x bank_address_16_18 %x enable_16k_mapping %x", addr, value, bank_address_16_18, enable_16k_mapping));
1639#endif
1640 return value;
1641 } else {
1642#ifdef LOG_READ_DF13
1643 LOG(("MMCREPLAY: DISABLED IO2 RD %04x", addr));
1644#endif
1645 }
1646 break;
1647 }
1648
1649 if (enable_io2) {
1650 mmcreplay_io2_device.io_source_valid = 1;
1651
1652 if (enable_ram_io) {
1653#ifdef LOG_READ_IO2_RAM
1654 LOG(("MMCREPLAY: RAM IO2 RD %04x %02x (%02x:%04x)", addr,
1655 mmcr_ram[(addr & 0x1fff) + (io2_ram_bank << 13)],
1656 io2_ram_bank, (addr & 0x1fff)));
1657#endif
1658 return mmcr_ram[(io2_ram_bank << 13) + 0x1f00 + (addr & 0xff)];
1659 }
1660#ifdef LOG_READ_IO2_ROM
1661 LOG(("MMCREPLAY: ROM IO2 RD %04x %02x (%02x:%04x)", addr,
1662 roml_banks[(addr & 0x1fff) + (io2_bank << 13)], io2_bank,
1663 (addr & 0x1fff)));
1664#endif
1665 return flash040core_read(flashrom_state, (io2_ram_bank << 13) + 0x1f00 + (addr & 0xff));
1666 } else {
1667#ifdef LOG_READ_IO2_DISABLED
1668 LOG(("MMCREPLAY: DISABLED IO2 RD %04x %02x (%02x:%04x)", addr,
1669 roml_banks[(io2_ram_bank << 13) + 0x1f00 + (addr & 0xff)],
1670 io2_bank, (addr & 0x1fff)));
1671#endif
1672 }
1673
1674 return 0;
1675}

1677void mmcreplay_io2_store(uint16_t addr, uint8_t value)
1678{
1679 switch (addr & 0xff) {
1680 case 0x10:
1681 /*
1682 * $DF10: MMC SPI transfer register
1683 *
1684 * byte written is sent to the card
1685 */
1686 if (enable_mmc_regs) {
1687#ifdef LOG_WRITE_DF10
1688 LOG(("MMCREPLAY: IO2 ST %04x %02x", addr, value));
1689#endif
1690 spi_mmc_data_write(value);
1691 return;
1692 }
1693 break;
1694 case 0x11:
1695 /*
1696 * $DF11: MMC control register
1697 * ------------------------
1698 * bit 0: 0 = MMC BIOS enabled, 1 = MMC BIOS disabled (R/W) (*)
1699 * bit 1: 0 = card selected, 1 = card not selected (R/W) (**)
1700 * bit 2: 0 = 250khz transfer, 1 = 8mhz transfer (R/W)
1701 * bit 3: ** ALWAYS 0 **
1702 * bit 4: ** ALWAYS 0 **
1703 * bit 5: 0 = allow RR rom when MMC BIOS disabled , 1 = disable RR ROM (R/W) (***)
1704 * bit 6: 0 = SPI write trigger mode, 1 = SPI read trigger mode (R/W)
1705 * bit 7: ** ALWAYS 0 **
1706 *
1707 * (*) Enabling MMC Bios sets ROM banking to the last 64K bank
1708 * (**) This bit also controls the green activity LED.
1709 * (***) When in mmcreplay bios mode, bit 5 controls RAM banking (0 = $e000 - $ffff, 1 = $8000 - $9fff)
1710 * When in 16K mode, bit 5 enables RAM at $a000 - $bfff
1711 */
1712 if (enable_mmc_regs) {
1713 disable_mmc_bios = (value) & 1; /* bit 0 */
1714 disable_rr_rom = (value >> 5) & 1; /* bit 5 */
1715
1716#ifdef LOG_WRITE_DF11
1717 LOG(("MMCREPLAY: IO2 ST %04x %02x disable_mmc_bios %x disable_rr_rom %x", addr, value, disable_mmc_bios, disable_rr_rom));
1718#endif
1719
1720 spi_mmc_card_selected_write((uint8_t)(((value >> 1) ^ 1) & 1)); /* bit 1 */
1721 spi_mmc_enable_8mhz_write((uint8_t)(((value >> 2)) & 1)); /* bit 2 */
1722 /* bit 3,4 always 0 */
1723 spi_mmc_trigger_mode_write((uint8_t)(((value >> 6)) & 1)); /* bit 6 */
1724 /* bit 7 always 0 */
1725 if (disable_mmc_bios) {
1726 /* if (enable_mmc_regs_pending) */
1727 {
1728 enable_mmc_regs = enable_mmc_regs_pending;
1729 }
1730 } else {
1731 /* clearing bit 0 resets write once for de01 */
1732 write_once = 0;
1733 }
1734 mmcreplay_update_mapper(CMODE_WRITE1, 0);
1735 return;
1736 }
1737 break;
1738 case 0x12:
1739 /*
1740 * $DF12: MMC status register
1741 * -----------------------
1742 * bit 0: 0 = SPI ready, 1 = SPI busy (R)
1743 * 1 = forbid ROM write accesses (W) (*)
1744 * bit 1: feedback of $DE00 bit 0 (GAME) (R)
1745 * bit 2: feedback of $DE00 bit 1 (EXROM) (R)
1746 * bit 3: 0 = card inserted, 1 = no card inserted (R)
1747 * bit 4: 0 = card write enabled, 1 = card write disabled (R)
1748 * bit 5: EEPROM DATA line / DDR Register (R/W) (**)
1749 * bit 6: 0 = RR compatibility mode, 1 = Extended mode (W) (***)
1750 * bit 7: EEPROM CLK line (W)
1751 *
1752 * (*) Setting this bit will disable writes to ROM until next reset.
1753 * (**) Setting DATA to "1" enables reading data bit from EEPROM at this position.
1754 * (***) Selecting RR compatibility mode limits RAM to 32K and
1755 * disables writes to extended banking register.
1756 * Selecting Extended mode enables full RAM banking and enables
1757 * Nordic Power mode in RR mode.
1758 */
1759 if (enable_mmc_regs) {
1760 /* FIXME bit 0: forbid write access to ROM */
1761
1762 disable_rom_write = (value ^ 1) & 1; /* bit 0 */
1763 enable_extended_mode = (value >> 6) & 1; /* bit 6 */
1764 if (enable_extended_mode) {
1765 rr_active = 1;
1766 }
1767 /* bit 2 ? FIXME */
1768 /* bit 3 ? FIXME */
1769#ifdef LOG_WRITE_DF12
1770 LOG(("MMCREPLAY: IO2 ST %04x %02x disable_rom_write %x enable_extended_mode %x", addr, value, disable_rom_write, enable_extended_mode));
1771#endif
1772
1773 /* EEPROM is only accessible in MMC Replay Bios mode */
1774 if (disable_mmc_bios == 0) {
1775 /*
1776 * bit 1: ddr FIXME
1777 * bit 4: status
1778 * bit 5: data/ddr
1779 * bit 7: clk
1780 */
1781 eeprom_port_write((uint8_t)((value >> 7) & 1), (uint8_t)((value >> 5) & 1),
1782 (value >> 1) & 1, (value >> 4) & 1);
1783 }
1784
1785 mmcreplay_update_mapper(CMODE_WRITE1, 0);
1786 return;
1787 }
1788 break;
1789 case 0x13:
1790 /*
1791 * $DF13: Extended banking register (*)
1792 * -----------------------------
1793 * bit 0: bank address 16 (R/W)
1794 * bit 1: bank address 17 (R/W)
1795 * bit 2: bank address 18 (R/W)
1796 * bit 3: ** ALWAYS 0 **
1797 * bit 4: ** ALWAYS 0 **
1798 * bit 5: 16K rom mapping (R/W)
1799 * bit 6: 1 = enable RR register (W)
1800 * bit 7: ** ALWAYS 0 **
1801 *
1802 * (*) Can only be read/written to when bit 6 of $DF12 is 1.
1803 */
1804
1805 if (enable_mmc_regs) {
1806 /* $df13 can only be READ AND WRITTEN if bit6 of $df12 = 1 */
1807 if (enable_extended_mode) {
1808 bank_address_16_18 = (value) & 7; /* bit 0-2 */
1809 enable_16k_mapping = (value >> 5) & 1; /* bit 5 */
1810 enable_rr_regs = (value >> 6) & 1; /* bit 6 */
1811 if ((value >> 6) & 1) { /* bit 6 */
1812 /* re-enable RR cartridge */
1813 rr_active = 1;
1814 }
1815#ifdef LOG_WRITE_DF13
1816 LOG(("MMCREPLAY: IO2 ST %04x %02x bank_address_16_18 %x enable_16k_mapping %x enable rr regs %x", addr, value, bank_address_16_18, enable_16k_mapping, ((value >> 6) & 1)));
1817#endif
1818 mmcreplay_update_mapper(CMODE_WRITE1, 0);
1819 return;
1820 }
1821 }
1822
1823 break;
1824 }
1825
1826 if (enable_io2) {
1827 if (enable_ram_io) {
1828#ifdef LOG_WRITE_IO2_RAM
1829 LOG(("MMCREPLAY: RAM IO2 ST %04x %02x (%02x:%04x)", addr, value,
1830 io2_ram_bank, (addr & 0x1fff)));
1831#endif
1832 mmcr_ram[(io2_ram_bank << 13) + 0x1f00 + (addr & 0xff)] = value;
1833 } else {
1834#ifdef LOG_WRITE_IO2_ROM
1835 LOG(("MMCREPLAY: ROM IO2 ST %04x %02x (%02x:%04x)",
1836 addr, value, io2_ram_bank, (addr & 0x1fff)));
1837#endif
1838 flash040core_store(flashrom_state, (io2_ram_bank << 13) + 0x1f00 + (addr & 0xff), value);
1839 }
1840 } else {
1841#ifdef LOG_WRITE_IO2_DISABLED
1842 LOG(("MMCREPLAY: DISABLED IO2 ST %04x %02x (%02x:%04x)", addr, value,
1843 io2_bank, (addr & 0x1fff)));
1844#endif
1845 }
1846}

2336

2343Disables any memory at $de00-$dfff due to CPLD routing problems. This is however
2344no problem, since RAM is always enabled at $8000-$9fff if you need it.

2347----------------
2348$8000 - $9fff: 8K RAM bank low
2349$e000 - $ffff: 8K RAM bank hi
