286static inline uint8_t io_read(io_source_list_t *list, uint16_t addr)
287{
288 io_source_list_t *current = list->next;
289 int io_source_counter = 0;
290 int io_source_valid = 0;
291 uint8_t realval = 0;
292 uint8_t retval = 0;
293 uint8_t firstval = 0;
294 unsigned int lowest_order = 0xffffffff;
295
296 while (current) {
297 if (current->device->read != NULL((void*)0)) {
298 if ((addr >= current->device->start_address) && (addr <= current->device->end_address)) {
299 retval = current->device->read((uint16_t)(addr & current->device->address_mask));
300 if (current->device->io_source_valid) {
301 /* high prio always overrides others, return immediatly */
302 if (current->device->io_source_prio == IO_PRIO_HIGH1) {
303 return retval;
304 }
305 if (io_source_valid == 0) {
306 /* on first valid read, initialize intermediate values */
307 firstval = realval = retval;
308 lowest_order = current->device->order;
309 /* do not count low prio, as it will always be overridden by others */
310 if (current->device->io_source_prio != IO_PRIO_LOW-1) {
311 io_source_counter++;
312 }
313 io_source_valid = 1;
314 } else {
315 /* ignore low prio reads when a real value is present already */
316 if (current->device->io_source_prio == IO_PRIO_LOW-1) {
317 retval = realval;
318 }
319 if (io_source_collision_handling == IO_COLLISION_METHOD_DETACH_LAST1) {
320 if (current->device->order < lowest_order) {
321 lowest_order = current->device->order;
322 realval = retval;
323 }
324 } else if (io_source_collision_handling == IO_COLLISION_METHOD_AND_WIRES2) {
325 realval &= retval;
326 }
327 /* do not count low prio, as it will always be overridden by others */
328 if (current->device->io_source_prio != IO_PRIO_LOW-1) {
329 /* if the nth read returns the same as the first read don't see it as a conflict */
330 if (retval != firstval) {
331 io_source_counter++;
332 }
333 }
334 }
335 }
336 }
337 }
338 current = current->next;
339 }
340
341 /* no valid I/O source was read, return phi1 */
342 if (io_source_valid == 0) {
343 return read_unused(addr);
344 }
345 /* only one valid I/O source was read, return value */
346 if (io_source_valid == 1) {
347 return firstval;
348 }
349 /* more than one I/O source was read, handle collision */
350 if (io_source_collision_handling == IO_COLLISION_METHOD_DETACH_ALL0) {
351 io_source_msg_detach_all(addr, io_source_counter, list);
352 return read_unused(addr);
353 } else if (io_source_collision_handling == IO_COLLISION_METHOD_DETACH_LAST1) {
354 io_source_msg_detach_last(addr, io_source_counter, list, lowest_order);
355 return realval;
356 } else if (io_source_collision_handling == IO_COLLISION_METHOD_AND_WIRES2) {
357 io_source_log_collisions(addr, io_source_counter, list);
358 return realval;
359 }
360 return read_unused(addr);
361}
