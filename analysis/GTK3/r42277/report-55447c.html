319static AVFrame *alloc_audio_frame(enum AVSampleFormat sample_fmt,
320 uint64_t channel_layout,
321 int sample_rate, int nb_samples)
322{
323 AVFrame *frame = VICE_P_AV_FRAME_ALLOC(*ffmpeglib.p_av_frame_alloc)();
324 int ret;
325
326 if (!frame) {
327 log_debug("ffmpegdrv: Error allocating an audio frame");
328 return NULL((void*)0);
329 }
330
331 frame->format = sample_fmt;
332 frame->channel_layout = channel_layout;
333 frame->sample_rate = sample_rate;
334 frame->nb_samples = nb_samples;
335
336 if (nb_samples) {
337 ret = VICE_P_AV_FRAME_GET_BUFFER(*ffmpeglib.p_av_frame_get_buffer)(frame, 0);
338 if (ret < 0) {
339 log_debug("ffmpegdrv: Error allocating an audio buffer");
340 return NULL((void*)0);
341 }
342 }
343
344 return frame;
345}

347static int ffmpegdrv_open_audio(AVFormatContext *oc, AVStream *st)
348{
349 AVCodecContext *c;
350 int audio_inbuf_samples;
351 int ret;
352 AVDictionary *opts = NULL((void*)0);
353
354 c = st->codec;
355 /* open codec */
356 /* aac encoder ist flaged 'experimental' */
357 VICE_P_AV_DICT_SET(*ffmpeglib.p_av_dict_set)(&opts, "strict", "experimental", 0);
358 ret = VICE_P_AVCODEC_OPEN2(*ffmpeglib.p_avcodec_open2)(c, avcodecaudio, &opts);
359 VICE_P_AV_DICT_FREE(*ffmpeglib.p_av_dict_free)(&opts);
360
361 if (ret < 0) {
362 log_debug("ffmpegdrv: could not open audio codec (%d)", ret);
363 return -1;
364 }
365
366 audio_is_open = 1;
367#ifdef HAVE_FFMPEG4
368 if (c->codec->capabilities & AV_CODEC_CAP_VARIABLE_FRAME_SIZE(1 << 16)) {
369#else
370 if (c->codec->capabilities & CODEC_CAP_VARIABLE_FRAME_SIZE) {
371#endif
372 audio_inbuf_samples = 10000;
373 } else {
374 audio_inbuf_samples = c->frame_size;
375 }
376
377 audio_st.frame = alloc_audio_frame(c->sample_fmt, c->channel_layout, c->sample_rate, audio_inbuf_samples);
378 audio_st.tmp_frame = alloc_audio_frame(AV_SAMPLE_FMT_S16, c->channel_layout, c->sample_rate, audio_inbuf_samples);
379
380 if (!audio_st.frame || !audio_st.tmp_frame) {
381 return -1;
382 }
383
384 ffmpegdrv_audio_in.size = audio_inbuf_samples * c->channels;
385 ffmpegdrv_audio_in.buffer = (int16_t *)audio_st.tmp_frame->data[0];
386 return 0;
387}

409static int ffmpegmovie_init_audio(int speed, int channels, soundmovie_buffer_t ** audio_in)
410{
411 AVCodecContext *c;
412 AVStream *st;
413 int i;
414
415 if (ffmpegdrv_oc == NULL((void*)0) || ffmpegdrv_fmt == NULL((void*)0)) {
416 return -1;
417 }
418
419 audio_init_done = 1;
420
421 if (ffmpegdrv_fmt->audio_codec == AV_CODEC_ID_NONE) {
422 return -1;
423 }
424
425 *audio_in = &ffmpegdrv_audio_in;
426
427 (*audio_in)->size = 0; /* not allocated yet */
428 (*audio_in)->used = 0;
429
430 st = VICE_P_AVFORMAT_NEW_STREAM(*ffmpeglib.p_avformat_new_stream)(ffmpegdrv_oc, avcodecaudio);
431 if (!st) {
432 log_debug("ffmpegdrv: Could not alloc audio stream");
433 return -1;
434 }
435
436 c = st->codec;
437
438 /* put sample parameters */
439 c->sample_fmt = avcodecaudio->sample_fmts ? avcodecaudio->sample_fmts[0] : AV_SAMPLE_FMT_S16;
440 c->bit_rate = audio_bitrate;
441 c->sample_rate = speed;
442
443 if (avcodecaudio->supported_samplerates) {
444 c->sample_rate = avcodecaudio->supported_samplerates[0];
445 for (i = 0; avcodecaudio->supported_samplerates[i]; i++) {
446 if (avcodecaudio->supported_samplerates[i] == speed)
447 c->sample_rate = speed;
448 }
449 }
450 c->channel_layout = (channels == 1 ? AV_CH_LAYOUT_MONO(0x00000004) : AV_CH_LAYOUT_STEREO(0x00000001|0x00000002));
451 c->channels = VICE_P_AV_GET_CHANNEL_LAYOUT_NB_CHANNELS(*ffmpeglib.p_av_get_channel_layout_nb_channels)(c->channel_layout);
452
453 st->time_base = (AVRational){ 1, c->sample_rate };
454 audio_st.st = st;
455 audio_st.next_pts = 0;
456 audio_st.samples_count = 0;
457
458 /* Some formats want stream headers to be separate. */
459 if (ffmpegdrv_oc->oformat->flags & AVFMT_GLOBALHEADER0x0040) {
460#ifdef HAVE_FFMPEG4
461 c->flags |= AV_CODEC_FLAG_GLOBAL_HEADER(1 << 22);
462#else

594static int ffmpegdrv_fill_rgb_image(screenshot_t *screenshot, AVFrame *pic)
595{
596 int x, y;
597 int dx, dy;
598 int colnum;
599 int bufferoffset;
600 int x_dim = screenshot->width;
601 int y_dim = screenshot->height;
602 int pix = 0;
603 /* center the screenshot in the video */
604 dx = (video_width - x_dim) / 2;
605 dy = (video_height - y_dim) / 2;
606 bufferoffset = screenshot->x_offset + (dx < 0 ? -dx : 0)
607 + (screenshot->y_offset + (dy < 0 ? -dy : 0)) * screenshot->draw_buffer_line_size;
608
609 for (y = 0; y < video_height; y++) {
610 for (x = 0; x < video_width; x++) {
611 colnum = screenshot->draw_buffer[bufferoffset + x];
612 pic->data[0][pix + 3*x] = screenshot->palette->entries[colnum].red;
613 pic->data[0][pix + 3*x + 1] = screenshot->palette->entries[colnum].green;
614 pic->data[0][pix + 3*x + 2] = screenshot->palette->entries[colnum].blue;
615 }
616 bufferoffset += screenshot->draw_buffer_line_size;
617 pix += pic->linesize[0];
618 }
619
620 return 0;
621}

710static void ffmpegdrv_init_video(screenshot_t *screenshot)
711{
712 AVCodecContext *c;
713 AVStream *st;
714
715 if (ffmpegdrv_oc == NULL((void*)0) || ffmpegdrv_fmt == NULL((void*)0)) {
716 return;
717 }
718

721 if (ffmpegdrv_fmt->video_codec == AV_CODEC_ID_NONE) {
722 return;
723 }
724

726 if (!st) {
727 log_debug("ffmpegdrv: Could not alloc video stream\n");
728 return;
729 }
730
731 c = st->codec;
732
733 /* put sample parameters */
734 c->bit_rate = video_bitrate;
735 /* resolution should be a multiple of 16 */
736 /* ffmpegdrv_fill_rgb_image only implements cutting so */
737 /* adding black border was removed */
738 video_width = c->width = screenshot->width & ~0xf;
739 video_height = c->height = screenshot->height & ~0xf;
740 /* frames per second */
741 st->time_base = VICE_P_AV_D2Q(*ffmpeglib.p_av_d2q)(machine_get_cycles_per_frame()
742 / (double)(video_halve_framerate ?
743 machine_get_cycles_per_second() / 2 :
744 machine_get_cycles_per_second()),
745 (1 << 16) - 1);
746 c->time_base = st->time_base;
747
748 c->gop_size = 12; /* emit one intra frame every twelve frames at most */
749 c->pix_fmt = AV_PIX_FMT_YUV420P;
750
751#if (LIBAVUTIL_VERSION_MICRO100 >= 100)
752 /* Avoid format conversion which would lead to loss of quality */
753 if (c->codec_id == AV_CODEC_ID_FFV1) {
754 c->pix_fmt = AV_PIX_FMT_0RGB32AV_PIX_FMT_BGR0;
755 }
756#endif
757
758 /* Use XVID instead of FMP4 FOURCC for better compatibility */
759 if (c->codec_id == AV_CODEC_ID_MPEG4) {
760 c->codec_tag = MKTAG('X', 'V', 'I', 'D')(('X') | (('V') << 8) | (('I') << 16) | ((unsigned
)('D') << 24));
761 }
762
763 /* Allow nonstandard framerates for MPEG1 codec */
764 if (c->codec_id == AV_CODEC_ID_MPEG1VIDEO) {
765 c->strict_std_compliance = FF_COMPLIANCE_EXPERIMENTAL-2;
766 }
767
768#ifdef HAVE_FFMPEG_SWSCALE
769 /* setup scaler */
770 if (c->pix_fmt != VICE_AV_PIX_FMT_RGB24AV_PIX_FMT_RGB24) {
771 sws_ctx = VICE_P_SWS_GETCONTEXT(*ffmpeglib.p_sws_getContext)
772 (video_width, video_height, VICE_AV_PIX_FMT_RGB24AV_PIX_FMT_RGB24,
773 video_width, video_height, c->pix_fmt,
774 SWS_BICUBIC4,
775 NULL((void*)0), NULL((void*)0), NULL((void*)0));
776 if (sws_ctx == NULL((void*)0)) {
777 log_debug("ffmpegdrv: Can't create Scaler!\n");
778 }
779 }
780#endif
781
782 video_st.st = st;
783 video_st.next_pts = 0;
784 framecounter = 0;
785
786 /* Some formats want stream headers to be separate. */
787 if (ffmpegdrv_oc->oformat->flags & AVFMT_GLOBALHEADER0x0040) {
788#ifdef HAVE_FFMPEG4
789 c->flags |= AV_CODEC_FLAG_GLOBAL_HEADER(1 << 22);
790#else
791 c->flags |= CODEC_FLAG_GLOBAL_HEADER;
792#endif
793 }
794
795 if (audio_init_done) {
796 ffmpegdrv_init_file();
797 }

927static int ffmpegdrv_record(screenshot_t *screenshot)
928{
929 AVCodecContext *c;
930 int ret;
931
932 if (audio_init_done && video_init_done && !file_init_done) {
933 ffmpegdrv_init_file();
934 }
935
936 if (video_st.st == NULL((void*)0) || !file_init_done) {
937 return 0;
938 }
939
940 if (audio_st.st && video_st.next_pts > audio_st.next_pts) {
941 /* drop this frame */
942 return 0;
943 }
944
945 framecounter++;
946 if (video_halve_framerate && (framecounter & 1)) {
947 /* drop every second frame */
948 return 0;
949 }
950
951 c = video_st.st->codec;
952
953 if (c->pix_fmt != VICE_AV_PIX_FMT_RGB24AV_PIX_FMT_RGB24) {

958 video_st.tmp_frame->data,
959 video_st.tmp_frame->linesize, 0, c->height,
960 video_st.frame->data, video_st.frame->linesize);
961 }
962 } else {
963 ffmpegdrv_fill_rgb_image(screenshot, video_st.frame);
964 }
965
966 video_st.frame->pts = video_st.next_pts++;
967
968#ifdef AVFMT_RAWPICTURE
969 if (ffmpegdrv_oc->oformat->flags & AVFMT_RAWPICTURE) {
970 AVPacket pkt;
971 VICE_P_AV_INIT_PACKET(*ffmpeglib.p_av_init_packet)(&pkt);
972 pkt.flags |= AV_PKT_FLAG_KEY0x0001;
973 pkt.stream_index = video_st.st->index;
974 pkt.data = (uint8_t*)video_st.frame;
975 pkt.size = sizeof(AVPicture);
976 pkt.pts = pkt.dts = video_st.frame->pts;
977
978 ret = VICE_P_AV_INTERLEAVED_WRITE_FRAME(*ffmpeglib.p_av_interleaved_write_frame)(ffmpegdrv_oc, &pkt);
979 } else
980#endif
981 {
982 AVPacket pkt = { 0 };
983 int got_packet;
984
985 VICE_P_AV_INIT_PACKET(*ffmpeglib.p_av_init_packet)(&pkt);
986
987 /* encode the image */
988 ret = VICE_P_AVCODEC_ENCODE_VIDEO2(*ffmpeglib.p_avcodec_encode_video2)(c, &pkt, video_st.frame, &got_packet);
989 if (ret < 0) {
990 log_debug("Error while encoding video frame");
991 return -1;
992 }
993 /* if zero size, it means the image was buffered */
994 if (got_packet) {
995 if (write_frame(ffmpegdrv_oc, &c->time_base, video_st.st, &pkt)<0)
996 {
997 log_debug("ffmpegdrv_encode_audio: Error while writing audio frame");
998 }
999 } else {
1000 ret = 0;
1001 }
1002 }
1003 if (ret < 0) {
1004 log_debug("Error while writing video frame");
1005 return -1;
1006 }
1007

1050 if (ffmpeg_drv.formatlist[i].audio_codecs != NULL((void*)0)) {
1051 lib_free(ffmpeg_drv.formatlist[i].audio_codecs);

1053 if (ffmpeg_drv.formatlist[i].video_codecs != NULL((void*)0)) {

1055 }
1056 i++;
1057 }
1058 lib_free(ffmpeg_drv.formatlist);
1059 }
1060 lib_free(ffmpeg_format);
1061}

1063static void ffmpeg_get_formats_and_codecs(void)
1064{
1065 int i, j, ai = 0, vi = 0, f;
1066 gfxoutputdrv_codec_t *audio_codec_list;
1067 gfxoutputdrv_codec_t *video_codec_list;
1068 gfxoutputdrv_codec_t *ac, *vc;
1069
1070 f = 0;
1071 ffmpegdrv_formatlist = lib_malloc(sizeof(gfxoutputdrv_format_t));
1072
1073 for (i = 0; formats_to_test[i].name != NULL((void*)0); i++) {
1074 if (VICE_P_AV_GUESS_FORMAT(*ffmpeglib.p_av_guess_format)(formats_to_test[i].name, NULL((void*)0), NULL((void*)0))) {
1075 audio_codec_list = NULL((void*)0);
1076 video_codec_list = NULL((void*)0);
1077 if (formats_to_test[i].audio_codecs != NULL((void*)0)) {
1078 ai = 0;
1079 audio_codec_list = lib_malloc(sizeof(gfxoutputdrv_codec_t));
1080 ac = formats_to_test[i].audio_codecs;
1081 for (j = 0; ac[j].name != NULL((void*)0); j++) {
1082 if ((ac[j].id == AV_CODEC_ID_NONE) || VICE_P_AVCODEC_FIND_ENCODER(*ffmpeglib.p_avcodec_find_encoder)(ac[j].id)) {
1083 audio_codec_list[ai++] = ac[j];
1084 audio_codec_list = lib_realloc(audio_codec_list, (ai + 1) * sizeof(gfxoutputdrv_codec_t));
1085 }
1086 }
1087 audio_codec_list[ai].name = NULL((void*)0);
1088 }
1089 if (formats_to_test[i].video_codecs != NULL((void*)0)) {
1090 vi = 0;
1091 video_codec_list = lib_malloc(sizeof(gfxoutputdrv_codec_t));
1092 vc = formats_to_test[i].video_codecs;
1093 for (j = 0; vc[j].name != NULL((void*)0); j++) {
1094 if (vc[j].id == AV_CODEC_ID_NONE || VICE_P_AVCODEC_FIND_ENCODER(*ffmpeglib.p_avcodec_find_encoder)(vc[j].id)) {
1095 video_codec_list[vi++] = formats_to_test[i].video_codecs[j];
1096 video_codec_list = lib_realloc(video_codec_list, (vi + 1) * sizeof(gfxoutputdrv_codec_t));
1097 }
1098 }
1099 video_codec_list[vi].name = NULL((void*)0);
1100 }
1101 if (((audio_codec_list == NULL((void*)0)) || (ai > 0)) && ((video_codec_list == NULL((void*)0)) || (vi > 0))) {
1102 ffmpegdrv_formatlist[f].name = lib_strdup(formats_to_test[i].name);
1103 ffmpegdrv_formatlist[f].audio_codecs = audio_codec_list;
1104 ffmpegdrv_formatlist[f++].video_codecs = video_codec_list;
1105 ffmpegdrv_formatlist = lib_realloc(ffmpegdrv_formatlist, (f + 1) * sizeof(gfxoutputdrv_format_t));
1106 }
1107 }
1108 }
1109 ffmpegdrv_formatlist[f].name = NULL((void*)0);
1110 ffmpeg_drv.formatlist = ffmpegdrv_formatlist;
1111}
