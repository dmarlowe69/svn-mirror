807 *
808 * Replace first occurences of \a search in \a original with \a replace.
809 *
810 * \param[in] original original string
811 * \param[in] search (sub)string of \a original to replace
812 * \param[in] replace replacement for \a search
813 *
814 * \return heap-allocated string
815 *
816 * \note free result with lib_free()
817 */
818static char *parser_strsubst(const char *original,
819 const char *search,
820 const char *replace)
821{
822 char *result = NULL((void*)0);
823
824 if (original == NULL((void*)0) || *original == '\0') {
825 /* empty string, return empty string */
826 result = lib_calloc(1, 1);
827 } else if (search == NULL((void*)0) || *search == '\0') {
828 /* no search string, return copy of original */
829 result = lib_strdup(original);

831 /* start scanning */
832 size_t slen = strlen(search);
833 size_t olen = strlen(original);
834
835 if (slen > olen) {
836 /* search string longer than original, will never find a match */
837 result = lib_strdup(original);
838 } else {
839 /* search and destroy! */
840 const char *pos = strstr(original, search);
841 if (pos == NULL((void*)0)) {
842 /* no match, return copy of original */
843 result = lib_strdup(original);
844 } else {
845 const char *opos; /* position in original */

848 size_t rlen = strlen(replace);
849

851 if (slen >= rlen) {

856 reslen = olen + (rlen - slen);

858

864 memcpy(rpos, original, pos - original)__builtin___memcpy_chk (rpos, original, pos - original, __builtin_object_size
(rpos, 0));

889 * string, in a case-insenstive manner.
890 *

904 return -1;

906 return 1;
907 }
908 }
909

911 /* more data, but equal for the requested first n chars */
912 return 0;
913 } else if (*s1 == '\0' && *s2 == '\0') {
914 /* no more data, strings match */
915 return 0;
916 } else if (*s1 == '\0') {
917 /* s2 is longer */
918 return 1;
919 } else {
920 return -1;
921 }
