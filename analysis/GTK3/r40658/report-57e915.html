337void vic_sound_clock(CLOCK cycles)
338{
339 CLOCK i;
340 int j, enabled;
341
342 if (cycles <= 0) {
343 return;
344 }
345
346 for (j = 0; j < 4; j++) {
347 int chspeed = "\4\3\2\1"[j];
348
349 if (snd.ch[j].ctr > cycles) {
350 snd.accum += snd.ch[j].out * cycles;
351 snd.ch[j].ctr -= cycles;
352 } else {
353 for (i = cycles; i; i--) {
354 snd.ch[j].ctr--;
355 if (snd.ch[j].ctr <= 0) {
356 int a = (~snd.ch[j].reg) & 127;
357 int edge_trigger;
358 a = a ? a : 128;
359 snd.ch[j].ctr += a << chspeed;
360 enabled = (snd.ch[j].reg & 128) >> 7;
361 edge_trigger = (noise_LFSR & 1) & !noise_LFSR0_old;
362
363 if((j != 3) || ((j == 3) && edge_trigger)) {
364 uint8_t shift = snd.ch[j].shift;
365 shift = ((shift << 1) | (((((shift & 128) >> 7)) ^ 1) & enabled));
366 snd.ch[j].shift = shift;
367 }
368 if(j == 3) {
369 int bit3 = (noise_LFSR >> 3) & 1;
370 int bit12 = (noise_LFSR >> 12) & 1;
371 int bit14 = (noise_LFSR >> 14) & 1;
372 int bit15 = (noise_LFSR >> 15) & 1;
373 int gate1 = bit3 ^ bit12;
374 int gate2 = bit14 ^ bit15;
375 int gate3 = (gate1 ^ gate2) ^ 1;
376 int gate4 = (gate3 & enabled) ^ 1;
377 noise_LFSR0_old = noise_LFSR & 1;
378 noise_LFSR = (noise_LFSR << 1) | gate4;
379 }
380 snd.ch[j].out = snd.ch[j].shift & (j == 3 ? enabled : 1);
381 }
382 snd.accum += snd.ch[j].out; /* FIXME: doesn't take DC offset into account */
383 }
384 }
385 }
386
387 snd.accum_cycles += cycles;
388}
