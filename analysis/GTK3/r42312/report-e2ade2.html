201/* Methods for randomly selecting bits to flip. */
202typedef enum {
203 RANDOM_METHOD_NONE, /* flip no (or all) bits */
204 RANDOM_METHOD_GEOM, /* generate bit intervals between flips */
205 RANDOM_METHOD_UNIFORM /* generate discrete uniform per bit */
206} random_method_t;
207
208/* Generate a random variate from the geometric distribution with success
209 * probability p, where the function parameter is the value log(1 - p). If each
210 * bit has a probability p of being flipped, this distribution corresponds to
211 * the number of non-flipped bits between each subsequent flipped bit. */
212static unsigned int random_method_geom_next(double log_1mp)
213{
214 /* For a uniform random variable U \in [0, 1], then X = log(U) / log(1 - p))
215 * is exponentially distributed with rate -log(1 - p), and floor(X) is
216 * geometrically distributed with success probability p. */
217 double u = lib_double_rand_unit();
218 /* u may be 0 but not 1; use 1 - u to avoid taking the log of 0. */
219 double g = floor(log1p(-u) / log_1mp);
220 /* Avoid overflow when converting to an unsigned int. */
221 return (g > (double)UINT_MAX(2147483647 *2U +1U)) ? UINT_MAX(2147483647 *2U +1U) : g;
222}

224/* this can be used to init arbitrary memory */
225void ram_init_with_pattern(uint8_t *memram, unsigned int ramsize, RAMINITPARAM *ramparam)
226{
227 unsigned int offset, j, k;
228 uint8_t value;
229
230 random_method_t random_method = RANDOM_METHOD_NONE;
231 unsigned int random_mask_initial = 0;
232 double log_1mp = -INFINITY__builtin_huge_valf();
233 unsigned int random_next = UINT_MAX(2147483647 *2U +1U);
234
235 if (ramparam->random_chance <= 0) {
236 /* flipping no bits */
237 random_method = RANDOM_METHOD_NONE;
238 random_mask_initial = 0x00;
239 } else if (ramparam->random_chance >= RANDOM_CHANCE_MAX0x1000) {
240 /* flipping all bits; same as no bits, but with the opposite mask */
241 random_method = RANDOM_METHOD_NONE;
242 random_mask_initial = 0xff;
243 } else if (ramparam->random_chance == (RANDOM_CHANCE_MAX0x1000 / 2)) {
244 /* flipping bits or not with equal probability; worst-case for the
245 * geometric spacing method, so handle separately */
246 random_method = RANDOM_METHOD_UNIFORM;
247 } else if (ramparam->random_chance < (RANDOM_CHANCE_MAX0x1000 / 2)) {
248 /* some other probability less than 0.5; generate the number of bits
249 * un-flipped between each flipped bit. */
250 random_method = RANDOM_METHOD_GEOM;
251 random_mask_initial = 0x00;
252 log_1mp = log1p((double)-ramparam->random_chance / RANDOM_CHANCE_MAX0x1000);
253 random_next = random_method_geom_next(log_1mp);
254 } else {
255 /* some other probability greater than 0.5; generate the number of bits
256 * flipped between each un-flipped bit. */
257 random_method = RANDOM_METHOD_GEOM;
258 random_mask_initial = 0xff;
259 log_1mp = log((double)ramparam->random_chance / RANDOM_CHANCE_MAX0x1000);
260 random_next = random_method_geom_next(log_1mp);
261 }
262
263 for (offset = 0; offset < ramsize; offset++) {
264
265 j = k = 0;
266 if (ramparam->value_invert) {
267 j = (((offset + ramparam->value_offset) / ramparam->value_invert) & 1) ? 0xff : 0x00;
268 }
269
270 if (ramparam->pattern_invert) {
271 k = ((offset / ramparam->pattern_invert) & 1) ? ramparam->pattern_invert_value : 0x00;
272 }
273
274 value = ramparam->start_value ^ j ^ k;
275
276 k = 0;
277 if (ramparam->random_start && ramparam->random_repeat) {
278 k = ((offset % ramparam->random_repeat) < ramparam->random_start) ? lib_unsigned_rand(0, 0xff) : 0;
279 }
280
281 j = 0;

291 while (random_next < 8) {
292 j ^= (1 << random_next);

296 break;
297 }

299 value ^= k ^ j;

301 memram[offset] = value;

313void ram_init_print_pattern(char *s, int len, char *eol)
314{
315 unsigned char *mem;
316 char *p = s, *pp;
317 int i, a;
318 const char hextab[16] = "0123456789abcdef";
319
320 mem = lib_malloc(len);
321 ram_init(mem, len);
322
323 for (a = 0; a < len;) {
324 /* add address at start if line */
325 *p++ = hextab[(a >> 12) & 0x0f];
326 *p++ = hextab[(a >> 8) & 0x0f];
327 *p++ = hextab[(a >> 4) & 0x0f];
328 *p++ = hextab[a & 0x0f];
329 *p++ = ':';
330 *p++ = ' ';
331 /* add 16 bytes hex dump */
332 for (i = 0; i < 16; i++, a++) {
333 *p++ = hextab[(mem[a] >> 4) & 0x0f];
334 *p++ = hextab[(mem[a]) & 0x0f];
335 *p++ = ' ';
336 }
337 /* add end of line */
338 pp = eol;
339 while (*pp) {
340 *p++ = *pp++;
341 }
342 /* after each full page add another end of line */
343 if ((a & 0xff) == 0) {
344 pp = eol;
345 while (*pp) {
346 *p++ = *pp++;
347 }
348 }
349 }
350 *p = 0;
351 lib_free(mem);
