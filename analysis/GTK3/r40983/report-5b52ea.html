321static AVFrame *alloc_audio_frame(enum AVSampleFormat sample_fmt,
322 uint64_t channel_layout,
323 int sample_rate, int nb_samples)
324{
325 AVFrame *frame = VICE_P_AV_FRAME_ALLOC(*ffmpeglib.p_av_frame_alloc)();
326 int ret;
327
328 if (!frame) {
329 log_debug("ffmpegdrv: Error allocating an audio frame");
330 return NULL((void*)0);
331 }
332
333 frame->format = sample_fmt;
334 frame->channel_layout = channel_layout;
335 frame->sample_rate = sample_rate;
336 frame->nb_samples = nb_samples;
337
338 if (nb_samples) {
339 ret = VICE_P_AV_FRAME_GET_BUFFER(*ffmpeglib.p_av_frame_get_buffer)(frame, 0);

341 log_debug("ffmpegdrv: Error allocating an audio buffer");
342 return NULL((void*)0);
343 }
344 }
345
346 return frame;

349static int ffmpegdrv_open_audio(AVFormatContext *oc, AVStream *st)
350{
351 AVCodecContext *c;
352 int audio_inbuf_samples;
353 int ret;
354 AVDictionary *opts = NULL((void*)0);
355
356 c = st->codec;
357 /* open codec */
358 /* aac encoder ist flaged 'experimental' */
359 VICE_P_AV_DICT_SET(*ffmpeglib.p_av_dict_set)(&opts, "strict", "experimental", 0);
360 ret = VICE_P_AVCODEC_OPEN2(*ffmpeglib.p_avcodec_open2)(c, avcodecaudio, &opts);
361 VICE_P_AV_DICT_FREE(*ffmpeglib.p_av_dict_free)(&opts);
362
363 if (ret < 0) {
364 log_debug("ffmpegdrv: could not open audio codec (%d)", ret);
365 return -1;
366 }
367
368 audio_is_open = 1;
369#ifdef HAVE_FFMPEG4
370 if (c->codec->capabilities & AV_CODEC_CAP_VARIABLE_FRAME_SIZE(1 << 16)) {
371#else
372 if (c->codec->capabilities & CODEC_CAP_VARIABLE_FRAME_SIZE) {
373#endif
374 audio_inbuf_samples = 10000;
375 } else {
376 audio_inbuf_samples = c->frame_size;
377 }
378
379 audio_st.frame = alloc_audio_frame(c->sample_fmt, c->channel_layout, c->sample_rate, audio_inbuf_samples);
380 audio_st.tmp_frame = alloc_audio_frame(AV_SAMPLE_FMT_S16, c->channel_layout, c->sample_rate, audio_inbuf_samples);
381
382 if (!audio_st.frame || !audio_st.tmp_frame) {
383 return -1;
384 }
385
386 ffmpegdrv_audio_in.size = audio_inbuf_samples * c->channels;
387 ffmpegdrv_audio_in.buffer = (int16_t *)audio_st.tmp_frame->data[0];
388 return 0;
389}

411static int ffmpegmovie_init_audio(int speed, int channels, soundmovie_buffer_t ** audio_in)
412{
413 AVCodecContext *c;
414 AVStream *st;
415 int i;
416
417 if (ffmpegdrv_oc == NULL((void*)0) || ffmpegdrv_fmt == NULL((void*)0)) {
418 return -1;
419 }
420
421 audio_init_done = 1;
422
423 if (ffmpegdrv_fmt->audio_codec == AV_CODEC_ID_NONE) {
424 return -1;
425 }
426
427 *audio_in = &ffmpegdrv_audio_in;
428
429 (*audio_in)->size = 0; /* not allocated yet */
430 (*audio_in)->used = 0;
431
432 st = VICE_P_AVFORMAT_NEW_STREAM(*ffmpeglib.p_avformat_new_stream)(ffmpegdrv_oc, avcodecaudio);
433 if (!st) {
434 log_debug("ffmpegdrv: Could not alloc audio stream");
435 return -1;
436 }
437
438 c = st->codec;
439
440 /* put sample parameters */
441 c->sample_fmt = avcodecaudio->sample_fmts ? avcodecaudio->sample_fmts[0] : AV_SAMPLE_FMT_S16;
442 c->bit_rate = audio_bitrate;
443 c->sample_rate = speed;
444
445 if (avcodecaudio->supported_samplerates) {
446 c->sample_rate = avcodecaudio->supported_samplerates[0];
447 for (i = 0; avcodecaudio->supported_samplerates[i]; i++) {
448 if (avcodecaudio->supported_samplerates[i] == speed)
449 c->sample_rate = speed;
450 }
451 }
452 c->channel_layout = (channels == 1 ? AV_CH_LAYOUT_MONO(0x00000004) : AV_CH_LAYOUT_STEREO(0x00000001|0x00000002));
453 c->channels = VICE_P_AV_GET_CHANNEL_LAYOUT_NB_CHANNELS(*ffmpeglib.p_av_get_channel_layout_nb_channels)(c->channel_layout);
454
455 st->time_base = (AVRational){ 1, c->sample_rate };
456 audio_st.st = st;
457 audio_st.next_pts = 0;
458 audio_st.samples_count = 0;
459
460 /* Some formats want stream headers to be separate. */
461 if (ffmpegdrv_oc->oformat->flags & AVFMT_GLOBALHEADER0x0040) {
462#ifdef HAVE_FFMPEG4
463 c->flags |= AV_CODEC_FLAG_GLOBAL_HEADER(1 << 22);
464#else
465 c->flags |= CODEC_FLAG_GLOBAL_HEADER;
466#endif
467 }

596static int ffmpegdrv_fill_rgb_image(screenshot_t *screenshot, AVFrame *pic)
597{
598 int x, y;
599 int dx, dy;
600 int colnum;
601 int bufferoffset;
602 int x_dim = screenshot->width;
603 int y_dim = screenshot->height;
604 int pix = 0;
605 /* center the screenshot in the video */
606 dx = (video_width - x_dim) / 2;
607 dy = (video_height - y_dim) / 2;
608 bufferoffset = screenshot->x_offset + (dx < 0 ? -dx : 0)
609 + (screenshot->y_offset + (dy < 0 ? -dy : 0)) * screenshot->draw_buffer_line_size;
610
611 for (y = 0; y < video_height; y++) {
612 for (x = 0; x < video_width; x++) {
613 colnum = screenshot->draw_buffer[bufferoffset + x];
614 pic->data[0][pix + 3*x] = screenshot->palette->entries[colnum].red;
615 pic->data[0][pix + 3*x + 1] = screenshot->palette->entries[colnum].green;
616 pic->data[0][pix + 3*x + 2] = screenshot->palette->entries[colnum].blue;
617 }
618 bufferoffset += screenshot->draw_buffer_line_size;
619 pix += pic->linesize[0];
620 }
621
622 return 0;
623}

712static void ffmpegdrv_init_video(screenshot_t *screenshot)
713{
714 AVCodecContext *c;
715 AVStream *st;
716
717 if (ffmpegdrv_oc == NULL((void*)0) || ffmpegdrv_fmt == NULL((void*)0)) {
718 return;
719 }
720
721 video_init_done = 1;
722
723 if (ffmpegdrv_fmt->video_codec == AV_CODEC_ID_NONE) {
724 return;
725 }
726
727 st = VICE_P_AVFORMAT_NEW_STREAM(*ffmpeglib.p_avformat_new_stream)(ffmpegdrv_oc, avcodecvideo);
728 if (!st) {
729 log_debug("ffmpegdrv: Could not alloc video stream\n");
730 return;
731 }
732
733 c = st->codec;
734
735 /* put sample parameters */
736 c->bit_rate = video_bitrate;
737 /* resolution should be a multiple of 16 */
738 /* ffmpegdrv_fill_rgb_image only implements cutting so */
739 /* adding black border was removed */
740 video_width = c->width = screenshot->width & ~0xf;
741 video_height = c->height = screenshot->height & ~0xf;
742 /* frames per second */
743 st->time_base = VICE_P_AV_D2Q(*ffmpeglib.p_av_d2q)(machine_get_cycles_per_frame()
744 / (double)(video_halve_framerate ?
745 machine_get_cycles_per_second() / 2 :
746 machine_get_cycles_per_second()),
747 (1 << 16) - 1);
748 c->time_base = st->time_base;
749
750 c->gop_size = 12; /* emit one intra frame every twelve frames at most */
751 c->pix_fmt = AV_PIX_FMT_YUV420P;
752
753#if (LIBAVUTIL_VERSION_MICRO100 >= 100)
754 /* Avoid format conversion which would lead to loss of quality */
755 if (c->codec_id == AV_CODEC_ID_FFV1) {
756 c->pix_fmt = AV_PIX_FMT_0RGB32AV_PIX_FMT_BGR0;
757 }
758#endif
759
760 /* Use XVID instead of FMP4 FOURCC for better compatibility */
761 if (c->codec_id == AV_CODEC_ID_MPEG4) {
762 c->codec_tag = MKTAG('X', 'V', 'I', 'D')(('X') | (('V') << 8) | (('I') << 16) | ((unsigned
)('D') << 24));
763 }
764
765 /* Allow nonstandard framerates for MPEG1 codec */
766 if (c->codec_id == AV_CODEC_ID_MPEG1VIDEO) {
767 c->strict_std_compliance = FF_COMPLIANCE_EXPERIMENTAL-2;
768 }
769
770#ifdef HAVE_FFMPEG_SWSCALE
771 /* setup scaler */
772 if (c->pix_fmt != VICE_AV_PIX_FMT_RGB24AV_PIX_FMT_RGB24) {
773 sws_ctx = VICE_P_SWS_GETCONTEXT(*ffmpeglib.p_sws_getContext)
774 (video_width, video_height, VICE_AV_PIX_FMT_RGB24AV_PIX_FMT_RGB24,
775 video_width, video_height, c->pix_fmt,
776 SWS_BICUBIC4,
777 NULL((void*)0), NULL((void*)0), NULL((void*)0));
778 if (sws_ctx == NULL((void*)0)) {
779 log_debug("ffmpegdrv: Can't create Scaler!\n");
780 }
781 }
782#endif
783
784 video_st.st = st;
785 video_st.next_pts = 0;
786 framecounter = 0;
787
788 /* Some formats want stream headers to be separate. */
789 if (ffmpegdrv_oc->oformat->flags & AVFMT_GLOBALHEADER0x0040) {
790#ifdef HAVE_FFMPEG4
791 c->flags |= AV_CODEC_FLAG_GLOBAL_HEADER(1 << 22);
792#else
793 c->flags |= CODEC_FLAG_GLOBAL_HEADER;
794#endif
795 }
796
797 if (audio_init_done) {
798 ffmpegdrv_init_file();
799 }
800}

929static int ffmpegdrv_record(screenshot_t *screenshot)
930{
931 AVCodecContext *c;
932 int ret;
933
934 if (audio_init_done && video_init_done && !file_init_done) {
935 ffmpegdrv_init_file();
936 }
937
938 if (video_st.st == NULL((void*)0) || !file_init_done) {
939 return 0;
940 }
941
942 if (audio_st.st && video_st.next_pts > audio_st.next_pts) {
943 /* drop this frame */
944 return 0;
945 }
946
947 framecounter++;
948 if (video_halve_framerate && (framecounter & 1)) {
949 /* drop every second frame */
950 return 0;
951 }
952
953 c = video_st.st->codec;
954
955 if (c->pix_fmt != VICE_AV_PIX_FMT_RGB24AV_PIX_FMT_RGB24) {
956 ffmpegdrv_fill_rgb_image(screenshot, video_st.tmp_frame);
957
958 if (sws_ctx != NULL((void*)0)) {
959 VICE_P_SWS_SCALE(*ffmpeglib.p_sws_scale)(sws_ctx,
960#if defined(STATIC_FFMPEG) || defined(SHARED_FFMPEG)
961 (const uint8_t * const *)video_st.tmp_frame->data,
962#else
963 video_st.tmp_frame->data,
964#endif
965 video_st.tmp_frame->linesize, 0, c->height,
966 video_st.frame->data, video_st.frame->linesize);
967 }
968 } else {
969 ffmpegdrv_fill_rgb_image(screenshot, video_st.frame);
970 }
971
972 video_st.frame->pts = video_st.next_pts++;
973
974#ifdef AVFMT_RAWPICTURE
975 if (ffmpegdrv_oc->oformat->flags & AVFMT_RAWPICTURE) {
976 AVPacket pkt;
977 VICE_P_AV_INIT_PACKET(*ffmpeglib.p_av_init_packet)(&pkt);
978 pkt.flags |= AV_PKT_FLAG_KEY0x0001;
979 pkt.stream_index = video_st.st->index;
980 pkt.data = (uint8_t*)video_st.frame;
981 pkt.size = sizeof(AVPicture);
982 pkt.pts = pkt.dts = video_st.frame->pts;
983
984 ret = VICE_P_AV_INTERLEAVED_WRITE_FRAME(*ffmpeglib.p_av_interleaved_write_frame)(ffmpegdrv_oc, &pkt);
985 } else
986#endif
987 {
988 AVPacket pkt = { 0 };
989 int got_packet;
990
991 VICE_P_AV_INIT_PACKET(*ffmpeglib.p_av_init_packet)(&pkt);
992
993 /* encode the image */
994 ret = VICE_P_AVCODEC_ENCODE_VIDEO2(*ffmpeglib.p_avcodec_encode_video2)(c, &pkt, video_st.frame, &got_packet);
995 if (ret < 0) {
996 log_debug("Error while encoding video frame");
997 return -1;
998 }
999 /* if zero size, it means the image was buffered */
1000 if (got_packet) {
1001 if (write_frame(ffmpegdrv_oc, &c->time_base, video_st.st, &pkt)<0)
1002 {
1003 log_debug("ffmpegdrv_encode_audio: Error while writing audio frame");
1004 }
1005 } else {
1006 ret = 0;
1007 }
1008 }
1009 if (ret < 0) {
1010 log_debug("Error while writing video frame");
1011 return -1;
1012 }
1013
1014 return 0;
1015}

1054 while (ffmpeg_drv.formatlist[i].name != NULL((void*)0)) {
1055 lib_free(ffmpeg_drv.formatlist[i].name);
1056 if (ffmpeg_drv.formatlist[i].audio_codecs != NULL((void*)0)) {
1057 lib_free(ffmpeg_drv.formatlist[i].audio_codecs);
1058 }
1059 if (ffmpeg_drv.formatlist[i].video_codecs != NULL((void*)0)) {
1060 lib_free(ffmpeg_drv.formatlist[i].video_codecs);
1061 }
1062 i++;
1063 }
1064 lib_free(ffmpeg_drv.formatlist);
1065 }
1066 lib_free(ffmpeg_format);
1067}

1069static void ffmpeg_get_formats_and_codecs(void)
1070{
1071 int i, j, ai = 0, vi = 0, f;
1072 gfxoutputdrv_codec_t *audio_codec_list;
1073 gfxoutputdrv_codec_t *video_codec_list;
1074 gfxoutputdrv_codec_t *ac, *vc;
1075
1076 f = 0;
1077 ffmpegdrv_formatlist = lib_malloc(sizeof(gfxoutputdrv_format_t));
1078
1079 for (i = 0; formats_to_test[i].name != NULL((void*)0); i++) {
1080 if (VICE_P_AV_GUESS_FORMAT(*ffmpeglib.p_av_guess_format)(formats_to_test[i].name, NULL((void*)0), NULL((void*)0))) {
1081 audio_codec_list = NULL((void*)0);
1082 video_codec_list = NULL((void*)0);
1083 if (formats_to_test[i].audio_codecs != NULL((void*)0)) {
1084 ai = 0;
1085 audio_codec_list = lib_malloc(sizeof(gfxoutputdrv_codec_t));
1086 ac = formats_to_test[i].audio_codecs;
1087 for (j = 0; ac[j].name != NULL((void*)0); j++) {
1088 if ((ac[j].id == AV_CODEC_ID_NONE) || VICE_P_AVCODEC_FIND_ENCODER(*ffmpeglib.p_avcodec_find_encoder)(ac[j].id)) {
1089 audio_codec_list[ai++] = ac[j];
1090 audio_codec_list = lib_realloc(audio_codec_list, (ai + 1) * sizeof(gfxoutputdrv_codec_t));
1091 }
1092 }
1093 audio_codec_list[ai].name = NULL((void*)0);
1094 }
1095 if (formats_to_test[i].video_codecs != NULL((void*)0)) {
1096 vi = 0;
1097 video_codec_list = lib_malloc(sizeof(gfxoutputdrv_codec_t));
1098 vc = formats_to_test[i].video_codecs;
1099 for (j = 0; vc[j].name != NULL((void*)0); j++) {
1100 if (vc[j].id == AV_CODEC_ID_NONE || VICE_P_AVCODEC_FIND_ENCODER(*ffmpeglib.p_avcodec_find_encoder)(vc[j].id)) {
1101 video_codec_list[vi++] = formats_to_test[i].video_codecs[j];
1102 video_codec_list = lib_realloc(video_codec_list, (vi + 1) * sizeof(gfxoutputdrv_codec_t));
1103 }
1104 }
1105 video_codec_list[vi].name = NULL((void*)0);
1106 }
1107 if (((audio_codec_list == NULL((void*)0)) || (ai > 0)) && ((video_codec_list == NULL((void*)0)) || (vi > 0))) {
1108 ffmpegdrv_formatlist[f].name = lib_strdup(formats_to_test[i].name);
1109 ffmpegdrv_formatlist[f].audio_codecs = audio_codec_list;
1110 ffmpegdrv_formatlist[f++].video_codecs = video_codec_list;
1111 ffmpegdrv_formatlist = lib_realloc(ffmpegdrv_formatlist, (f + 1) * sizeof(gfxoutputdrv_format_t));
1112 }
1113 }
1114 }
1115 ffmpegdrv_formatlist[f].name = NULL((void*)0);
1116 ffmpeg_drv.formatlist = ffmpegdrv_formatlist;
1117}
